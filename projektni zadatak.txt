#include<bits/stdc++.h>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int p;
    cin >> p;

    vector<pair<int, int>> a(p); 
    for (int i = 0; i < p; ++i) {
        cin >> a[i].first >> a[i].second;
    }

    int v;
    cin >> v;

    vector<int> vrece(v);
    for (int i = 0; i < v; ++i) {
        cin >> vrece[i];
    }

    
    sort(a.begin(), a.end());
    
    sort(vrece.begin(), vrece.end());

    int zbir = 0;
    int indeks = 0;
    
    priority_queue<int> pq;

    for (int i = 0; i < v; ++i) {
        int b = vrece[i];

        
        while (indeks < p && a[indeks].first <= b) {
            pq.push(a[indeks].second);
            ++indeks;
        }

        
        if (!pq.empty()) {
            zbir += pq.top();
            pq.pop();
        }
    }

    cout << zbir << '\n';

    return 0;
}
